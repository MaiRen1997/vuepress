JS的模块化

# js中的模块化

# es6语法

  1. 导出方式

    tips: export default 在每个模块中只能使用一次，否则报错

    // 默认导出
let a = 2;
let b = 3;
function show() { }
export  default  {a,b,show}
// 按需导出
export let a = 10
export let b = 'aaa'
export function say = function() {  }

  2. 导入方式

    // 默认导入
import card from  '/components/card.vue'
// 按需导入
import { a,b,say }  from  '模块标识符(路径)'

  3. 有时候只是想执行某模块的代码，并不需要其中向外暴露的成员，此时可以直接导入并执行模块代码。

    import  './index.js'

# commonJS语法

理论知识：

（1）每个文件都是一个模块，有自己的作用域模块内部的变量、方法或者类都是私有的，其他文件访问不到； （2）每个模块内部都有两个变量可使用，require 和 module; （3）【module】是一个储存了模块信息的对象，

let module = {
  moduleName: moduleName,
  exports: {},
  loaded: true,
}

（4）【exports】Node.js为方便起见，给每个模块增加了一个私有变量 exports，指向module.exports,相当于在每个模块开头加了一句var exports = module.exports。exports 是 module 上的一个属性，储存了模块暴露的接口或属性，使用require 加载的某个模块的值就是该模块 exports 导出的值；

如果模块只导出一个值，可以使用 module.exports = a 不可以使用 exports = a 因为这样会切换exports与 module.exports 的联系。

（5）【require】require函数的功能是加载并执行模块，然后返回该模块的exports对象。模块第一次加载后会被缓存，之后直接取出缓存中该模块的modules.exports属性。

使用exports导出， 使用require引入

// a.js
var name = 'morrain'
var age = 18
exports.name = name
exports.getAge = function () {
  return age
}

// b.js
var a = require('a.js')
console.log('a.name=', a.name)
console.log('a.age=', a.getAge())

//module1.js
module.exports = {
  msg: 'module1',
  fn () {
    console.log(this.msg)
  }
}
// module2.js
//module.exports 一个文件只能出现一次，要不容易被下面的覆盖
module.exports = function () {
  console.log('module2')
}
// module3.js
exports.fn = function() {
  console.log('fn() module3')
}
exports.bar = function () {
  console.log('bar() module3')
}
// app.js
let module1 = require('./modules/module1')
let module2 = require('./modules/module2')
let module3 = require('./modules/module3')

module1.fn()
module2();
module3.fn()
module3.bar()

# AMD语法

// file1文件
//没有依赖模块
define(function (){
  let name = 'dataService.js'
  function getName() {
    return name
  }
  //暴露模块
  return { getName }
}
  );
// file2文件
//定义有依赖的模块
define([
  'dataService'
], function(dataService) {
  let msg = 'alert.js';
  function showMsg() {
    console.log(msg, dataService.getName())
  }
  return {showMsg}
});
// main.js文件
(function() {
// requirejs配置文件配置
  requirejs.config({
    baseUrl: 'js/', //基本的路径，从根目录开始
    paths: {
      dataService: './modules/dataService',
      alerter: './modules/alerter'
    }
  })
//
  requirejs(['alerter'], function(alerter){
    alerter.showMsg()
  })
})()
